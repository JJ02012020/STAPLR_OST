# Install required packages using pip

pip install pandas         # For handling and analyzing data
pip install matplotlib     # For creating visualizations and plots
pip install seaborn        # For statistical data visualization and aesthetics
pip install customtkinter  # For modern, themed Tkinter-based GUI
pip install pyttsx3        # For enabling text-to-speech functionality
pip install python-docx    # For reading and writing Microsoft Word documents
pip install ollama         # For AI and language model operations
